5.2 Scatter Plot

ggplot(
  data = urban_noise,
  aes(x = temperature_c, y = decibel_level)
) +
  geom_point(alpha = 0.6, size = 2, colour = "blue") +
  geom_smooth(colour = "gray") +
  labs(
    title = "Relationship Between Temperature and Noise Level",
    x = "Temperature (°C)",
    y = "Noise Level (dB)"
  ) +
  theme_minimal()
`geom_smooth()` using method = 'gam' and formula = 'y ~ s(x, bs = "cs")'

5.3 Boxplot

ggplot(
  data = urban_noise,
  aes(x = factor(is_weekend), y = decibel_level, fill = factor(is_weekend))
) +
  geom_boxplot() +
  labs(
    title = "Noise Levels on Weekends vs Weekdays",
    x = "Weekend (1 = Yes, 0 = No)",
    y = "Noise Level (dB)"
  ) +
  theme_light() +
  theme(legend.position = "none")
